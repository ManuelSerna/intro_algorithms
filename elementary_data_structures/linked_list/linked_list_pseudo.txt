//***********************************************
 Linked list pseudo code.
//***********************************************

//-----------------------------------------------
 Search linked list.
//-----------------------------------------------
LIST_SEARCH(L, k)
    x = L.head
    while x != NIL and x.key != k
        x = x.next
    return x

//-----------------------------------------------
 Insert a value to the linked list.
//-----------------------------------------------
LIST_INSERT(L, x)
    x.next = L.head
    if L.head != NIL
        L.head.prev = x
    L.head = x
    x.prev = NIL

//-----------------------------------------------
 Delete a value from the linked list.
//-----------------------------------------------
LIST_DELETE(L, x)
    if x.prev != NIL
        x.prev.next = x.next
    else
        L.head = x.next
    
    if x.next != NIL
        x.next.prev = x.prev
